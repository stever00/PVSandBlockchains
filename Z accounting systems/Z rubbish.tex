\begin{schema}{A}
a : \seq T
\where
(a(1)).a1 = 1\\
(a(1)).a2 = 2
\end{schema}

\begin{schema}{B}
b : \seq S
\end{schema}

\begin{schema}{T}
a1 : \nat\\
a2 : \nat
\end{schema}

\begin{schema}{S}
b1 : \nat\\
b2 : \nat
\end{schema}

\begin{schema}{RTS}
T\\
S\\
\where
a1 = b1 + 2\\
a2 = b2 + 3
\end{schema}

\begin{zed}
RTShideA \defs [S | \exists T @ RTS]\\
RTShideB \defs [T | \exists S @ RTS]
\end{zed}
